    TreeNode node=stack.pop();
    if (node.right!=null) {
        pushLeft(node.right);
    }

public boolean hasNext() {
    return !stack.isEmpty();
}

public void pushLeft(TreeNode node){
    while(node!=null){
        stack.add(node);
        node=node.left;
    }
